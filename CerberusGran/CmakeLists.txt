cmake_minimum_required(VERSION 3.15)

project(CerberusGran VERSION 0.1.0)

set(JUCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../JUCE")

add_subdirectory(${JUCE_DIR} JUCE)

# Build universal binaries so REAPER can load in native or Rosetta modes.
if(APPLE)
    set(CMAKE_OSX_ARCHITECTURES "arm64;x86_64" CACHE STRING "macOS architectures" FORCE)
endif()

# Ensure plugin copy runs even if JUCE cache var was previously configured OFF.
set(JUCE_COPY_PLUGIN_AFTER_BUILD ON CACHE BOOL "Copy plugin bundles after build" FORCE)

juce_add_plugin(CerberusGran
    COMPANY_NAME "Ishaan"
    IS_SYNTH FALSE
    NEEDS_MIDI_INPUT FALSE
    NEEDS_MIDI_OUTPUT FALSE
    IS_MIDI_EFFECT FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    PLUGIN_MANUFACTURER_CODE ISHN
    PLUGIN_CODE CBRG
    FORMATS VST3
    PRODUCT_NAME "CerberusGran"
)

target_sources(CerberusGran PRIVATE
    Source/PluginProcessor.cpp
    Source/PluginProcessor.h
    Source/PluginEditor.cpp
    Source/PluginEditor.h
    Source/RingBuffer.h
    Source/Grain.h
    Source/WindowFunctions.h
    Source/GrainHead.h
    Source/GrainHead.cpp
    Source/GrainEngine.h
    Source/GrainEngine.cpp
    Source/Bitcrusher.h
    Source/Parameters.h
)

target_compile_definitions(CerberusGran PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(CerberusGran PRIVATE
    juce::juce_audio_utils
    juce::juce_dsp
    juce::juce_gui_extra
)

juce_generate_juce_header(CerberusGran)
